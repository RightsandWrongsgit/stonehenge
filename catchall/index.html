Error: Service not found for !

Service is not probably not defined in any docker-compose.yml file labels for Traefik.

You should have something like this on your project:


version: '3.7'

services:
  yourproject:
    image: someimage
    container_name: "${COMPOSE_PROJECT_NAME}-yourproject"
    labels:
      traefik.enable: true
      traefik.port: 80
      traefik.frontend.rule: "Host:"
    networks:
      - dev-network

networks:

  # Network from druidfi/stonehenge
  dev-network:
    name: stonehenge-network
    external: true
